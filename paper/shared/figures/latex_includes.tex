% === Fig 1: Hero figure (method + cross-material comparison) ===
% NOTE: Method overview panel is a MANUAL architecture diagram.
%       Right panel reuses fig2_main_comparison.
\begin{figure*}[t]
    \centering
    \includegraphics[width=0.98\textwidth]{figures/fig1_hero.pdf}
    \caption{\textbf{Probe-Then-Act framework and cross-material performance.} Left: Two-stage architecture --- (i) the robot executes a short probing phase, (ii) a belief encoder maps the probe trace to a latent material representation $z$ with uncertainty $\sigma$, (iii) the task policy is conditioned on $(s_t, z, \sigma)$ to produce adaptive joint-space residual actions. Right: Transfer efficiency across five material splits. Our method (M7) achieves a statistically significant improvement on elastoplastic materials (\textbf{$^*p<0.05$}) while remaining competitive on other out-of-distribution variants. Error bars: $\pm 1$ std over 3 seeds.}
    \label{fig:hero}
\end{figure*}

% === Fig 2: Main cross-material comparison ===
\begin{figure}[t]
    \centering
    \includegraphics[width=0.98\columnwidth]{figures/fig2_main_comparison.pdf}
    \caption{Cross-material transfer efficiency comparison. Probe-Then-Act (M7) delivers a 14.7pp absolute (32\% relative) improvement over the reactive baseline (M1) on elastoplastic materials ($^*$), demonstrating that active probing uncovers actionable information for viscoelastic media. On simpler splits the reactive baseline remains competitive, indicating the method operates in a material-specific regime. Error bars: $\pm 1$ std over 3 seeds.}
    \label{fig:main}
\end{figure}

% === Fig 3: Ablation study ===
\begin{figure*}[t]
    \centering
    \includegraphics[width=0.98\textwidth]{figures/fig3_ablation.pdf}
    \caption{\textbf{Ablation study on OOD Elastoplastic.} (a) Removing the probe phase drops transfer by 27.6pp; removing the belief encoder drops it by 14.4pp, indicating both modules are necessary for the observed adaptation. (b) Spill ratio follows the same ordering. (c) Success rate closely tracks transfer efficiency. Error bars: $\pm 1$ std over 3 seeds.}
    \label{fig:ablation}
\end{figure*}

% === Fig 4: Per-seed distribution ===
\begin{figure}[t]

    \label{fig:seeds}
\end{figure}
