% === Fig 3: Ablation study ===
\begin{figure*}[t]
    \centering
    \includegraphics[width=0.98\textwidth]{figures/fig3_ablation.pdf}
    \caption{\textbf{Ablation study on OOD Elastoplastic.} (a) Removing the probe phase drops mean transfer by 27.6pp and falls below M1; removing the belief encoder drops 14.4pp from full \PTA{} and returns to roughly M1-level transfer. (b) Spill ratio follows the same ordering. (c) Success rate tracks transfer efficiency. Error bars: $\pm 1$ seed standard deviation.}

\end{figure*}
