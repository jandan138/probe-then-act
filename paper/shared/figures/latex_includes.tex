% === Fig 1: Hero figure (method + cross-material comparison) ===
% NOTE: Method overview panel is a MANUAL architecture diagram.
%       Right panel reuses fig2_main_comparison.
\begin{figure*}[t]
    \centering
    \includegraphics[width=0.98\textwidth]{figures/fig1_hero.pdf}
    \caption{\textbf{Probe-Then-Act framework and cross-material performance.} Left: Two-stage architecture --- (i) the robot executes a short probing phase, (ii) an encoder maps the probe trace to a latent task-conditioning vector $z$ with learned scale $\sigma$, (iii) the task policy is conditioned on $(s_t, z, \sigma)$ to produce joint-space residual actions. Right: Transfer-efficiency deltas across five material splits. M7 improves mean transfer on elastoplastic and underperforms on the four other splits; the comparison is descriptive, not significance-tested. Error bars: $\pm 1$ seed standard deviation.}
    \label{fig:hero}
\end{figure*}

% === Fig 2: Main cross-material comparison ===
\begin{figure}[t]
    \centering
    \includegraphics[width=0.98\columnwidth]{figures/fig2_main_comparison.pdf}
    \caption{Cross-material transfer efficiency comparison. Probe-Then-Act (M7) delivers a 14.7pp absolute mean improvement over the reactive baseline (M1) on the elastoplastic split and underperforms on the other four splits. The result is directional and descriptive rather than statistically conclusive. Error bars: $\pm 1$ seed standard deviation.}
    \label{fig:main}
\end{figure}

% === Fig 3: Ablation study ===
\begin{figure*}[t]
    \centering
    \includegraphics[width=0.98\textwidth]{figures/fig3_ablation.pdf}
    \caption{\textbf{Ablation study on OOD Elastoplastic.} (a) Removing the probe phase drops mean transfer by 27.6pp; removing the belief encoder drops it by 14.4pp, suggesting both components contribute to the observed split-specific gain. (b) Spill ratio follows the same ordering. (c) Success rate tracks transfer efficiency. Error bars: $\pm 1$ seed standard deviation.}
    \label{fig:ablation}
\end{figure*}

% === Fig 4: Per-seed distribution ===
\begin{figure}[t]
    \centering
    \includegraphics[width=0.98\columnwidth]{figures/fig4_seed_distribution.pdf}
    \caption{Per-seed transfer efficiency for Probe-Then-Act (M7) versus reactive baseline (M1). Bar heights show the 3-seed mean; overlaid points show individual seeds. Per-seed dispersion is largest on elastoplastic, where M7 has a higher mean despite disagreement across seeds.}
    \label{fig:seeds}
\end{figure}
